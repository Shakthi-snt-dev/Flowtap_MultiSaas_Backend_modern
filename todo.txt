# Standalone industry APIs — each creates full schema for its own DB

# Food
dotnet ef migrations add InitialFood --context FoodDbContext `
  --project IndustryModules/Flowtap_Food `
  --startup-project APIEndpoints/Flowtap_Food_API `
  --output-dir Persistence/Migrations
